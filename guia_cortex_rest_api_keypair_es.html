import json
import asyncio
import time
import hashlib
import base64
import aiohttp
import pandas as pd
import jwt
from typing import Optional
from pydantic import BaseModel
from cryptography.hazmat.primitives import serialization
from _snowflake import get_generic_secret_string, vectorized

_ACCOUNT = "aikoqom-aqb12336"
_USER = "DSANCHEZ"
_BASE_URL = f"https://{_ACCOUNT}.snowflakecomputing.com/api/v2/cortex/v1"
_MESSAGES_ENDPOINT = f"{_BASE_URL}/messages"
_CHAT_ENDPOINT = f"{_BASE_URL}/chat/completions"

_MAX_BATCH_ROWS = 20
_MAX_RETRIES = 8
_BASE_DELAY = 2.0
_MAX_DELAY = 60.0
_CONCURRENCY_LIMIT = 3
_JWT_TTL_SECONDS = 60


def _generate_jwt(private_key_pem: str) -> str:
    account_upper = _ACCOUNT.upper()
    user_upper = _USER.upper()
    qualified_user = f"{account_upper}.{user_upper}"

    private_key = serialization.load_pem_private_key(
        private_key_pem.encode('utf-8'),
        password=None,
    )

    public_key_der = private_key.public_key().public_bytes(
        serialization.Encoding.DER,
        serialization.PublicFormat.SubjectPublicKeyInfo,
    )
    sha256_hash = hashlib.sha256(public_key_der).digest()
    fingerprint = "SHA256:" + base64.b64encode(sha256_hash).decode('utf-8')

    now = int(time.time())
    payload = {
        "iss": f"{qualified_user}.{fingerprint}",
        "sub": qualified_user,
        "iat": now,
        "exp": now + _JWT_TTL_SECONDS,
    }
    return jwt.encode(payload, private_key, algorithm="RS256")


class BatchCompletionResult(BaseModel):
    success: bool
    response: Optional[str] = None
    error: Optional[str] = None
    model: Optional[str] = None
    usage: Optional[dict] = None
    retries: Optional[int] = None

    def to_dict(self) -> dict:
        return self.model_dump(exclude_none=True)

    @classmethod
    def success_result(cls, response, usage, model, retries=0):
        return cls(success=True, response=response, model=model, usage=usage, retries=retries)

    @classmethod
    def error_result(cls, error, retries=0):
        return cls(success=False, error=error, retries=retries)

    @classmethod
    def empty_result(cls):
        return cls(success=True, response="")


class CortexAsyncClient:
    def __init__(self, token: str):
        self.token = token

    @staticmethod
    def _is_claude(model: str) -> bool:
        return "claude" in model.lower()

    def _get_headers(self, is_claude: bool) -> dict:
        headers = {
            "Authorization": f"Bearer {self.token}",
            "Content-Type": "application/json",
            "Accept": "application/json",
            "X-Snowflake-Authorization-Token-Type": "KEYPAIR_JWT",
        }
        if is_claude:
            headers["anthropic-version"] = "2023-06-01"
        return headers

    @staticmethod
    def _parse_response_format(fmt_str):
        if pd.isna(fmt_str) or str(fmt_str).strip() in ('None', 'nan', ''):
            return None
        try:
            return json.loads(fmt_str)
        except json.JSONDecodeError:
            return {"type": str(fmt_str)}

    @staticmethod
    def _normalize_usage(usage: dict, is_claude: bool) -> dict:
        if not is_claude:
            return usage
        return {
            "prompt_tokens": usage.get("input_tokens", 0),
            "completion_tokens": usage.get("output_tokens", 0),
            "total_tokens": usage.get("input_tokens", 0) + usage.get("output_tokens", 0),
            "prompt_tokens_details": {
                "cached_tokens": usage.get("cache_read_input_tokens", 0)
            },
            "cache_creation_input_tokens": usage.get("cache_creation_input_tokens", 0),
            "cache_read_input_tokens": usage.get("cache_read_input_tokens", 0),
        }

    @staticmethod
    def _parse_content(result: dict, is_claude: bool) -> str:
        if is_claude:
            for block in result.get("content", []):
                if block.get("type") == "text":
                    return block.get("text", "")
            return ""
        return result.get("choices", [{}])[0].get("message", {}).get("content", "")

    async def _fetch_with_retry(self, session, semaphore, payload, headers, endpoint, row_idx, model, is_claude):
        retries = 0
        delay = _BASE_DELAY

        for attempt in range(_MAX_RETRIES + 1):
            async with semaphore:
                try:
                    async with session.post(endpoint, json=payload, headers=headers) as resp:
                        if resp.status == 200:
                            result = await resp.json()
                            content = self._parse_content(result, is_claude)
                            usage = self._normalize_usage(result.get("usage", {}), is_claude)
                            return row_idx, BatchCompletionResult.success_result(content, usage, model, retries)

                        if resp.status == 429 or resp.status >= 500:
                            retries += 1
                            if attempt < _MAX_RETRIES:
                                retry_after = resp.headers.get("Retry-After")
                                if retry_after:
                                    delay = min(float(retry_after), _MAX_DELAY)
                                pass
                            else:
                                error_text = await resp.text()
                                return row_idx, BatchCompletionResult.error_result(
                                    f"HTTP_{resp.status} after {retries} retries: {error_text[:200]}", retries
                                )
                        else:
                            error_text = await resp.text()
                            return row_idx, BatchCompletionResult.error_result(f"HTTP_{resp.status}: {error_text[:200]}", retries)

                except asyncio.TimeoutError:
                    retries += 1
                    if attempt >= _MAX_RETRIES:
                        return row_idx, BatchCompletionResult.error_result(f"Timeout after {retries} retries", retries)
                except Exception as e:
                    return row_idx, BatchCompletionResult.error_result(str(e)[:200], retries)

            await asyncio.sleep(delay)
            delay = min(delay * 2, _MAX_DELAY)

        return row_idx, BatchCompletionResult.error_result("Exhausted retries", _MAX_RETRIES)

    def _build_payload_messages_api(self, model, sys_prompt, user_prompt):
        payload = {
            "model": model,
            "max_tokens": 4096,
            "temperature": 0,
            "stream": False,
            "messages": [{"role": "user", "content": user_prompt}],
        }
        if sys_prompt:
            payload["system"] = [
                {"type": "text", "text": sys_prompt, "cache_control": {"type": "ephemeral"}}
            ]
        return payload

    def _build_payload_chat_api(self, model, sys_prompt, user_prompt, resp_fmt):
        messages = []
        if sys_prompt:
            messages.append({"role": "system", "content": sys_prompt})
        messages.append({"role": "user", "content": user_prompt})
        payload = {"model": model, "messages": messages, "temperature": 0, "stream": False}
        if resp_fmt:
            payload["response_format"] = resp_fmt
        return payload

    async def process_dataframe(self, df):
        if df.empty:
            return []

        model = str(df.iloc[0, 0])
        sys_prompt = str(df.iloc[0, 1]) if pd.notna(df.iloc[0, 1]) else ""
        has_fmt_col = df.shape[1] > 3
        is_claude = self._is_claude(model)
        endpoint = _MESSAGES_ENDPOINT if is_claude else _CHAT_ENDPOINT
        headers = self._get_headers(is_claude)

        semaphore = asyncio.Semaphore(_CONCURRENCY_LIMIT)
        connector = aiohttp.TCPConnector(limit=_MAX_BATCH_ROWS, ssl=False)
        timeout = aiohttp.ClientTimeout(total=180)

        async with aiohttp.ClientSession(connector=connector, timeout=timeout) as session:
            tasks = []
            for i, row in df.iterrows():
                user_prompt = str(row[2]) if pd.notna(row[2]) else ""
                if not user_prompt:
                    tasks.append(asyncio.sleep(0, result=(i, BatchCompletionResult.empty_result())))
                    continue

                if is_claude:
                    payload = self._build_payload_messages_api(model, sys_prompt, user_prompt)
                else:
                    resp_fmt = None
                    if has_fmt_col:
                        resp_fmt = self._parse_response_format(row[3])
                    payload = self._build_payload_chat_api(model, sys_prompt, user_prompt, resp_fmt)

                tasks.append(self._fetch_with_retry(session, semaphore, payload, headers, endpoint, i, model, is_claude))

            results = await asyncio.gather(*tasks)
            return [res[1].to_dict() for res in sorted(results, key=lambda x: x[0])]


@vectorized(input=pd.DataFrame, max_batch_size=_MAX_BATCH_ROWS)
def complete_batch(df):
    private_key_pem = get_generic_secret_string('private_key')
    token = _generate_jwt(private_key_pem)
    client = CortexAsyncClient(token)
    results = asyncio.run(client.process_dataframe(df))
    return pd.Series(results)

    GROUP BY modelo
    UNION ALL
    SELECT
        'AI_COMPLETE Native' AS metodo, modelo,
        COUNT(*) AS total_rows,
        SUM(prompt_tokens) AS total_prompt_tokens,
        NULL AS total_cached_tokens,
        SUM(completion_tokens) AS total_output_tokens,
        ROUND(SUM(input_cost_credits) * 4, 4) AS input_cost,
        NULL AS cache_cost,
        ROUND(SUM(output_cost_credits) * 4, 4) AS output_cost,
        ROUND(SUM(total_cost_usd), 4) AS total_cost_usd
    FROM CORTEX_UTILS.PUBLIC.LLAMADAS_ANALISIS_NATIVE_LLAMA
    GROUP BY modelo

ORDER BY total_cost_usd;

-- 10b.3 Mega-comparativa: 4 métodos (Claude REST, Claude Native, Llama REST, Llama Native)
SELECT
    metodo, modelo, total_rows,
    total_prompt_tokens, total_cached_tokens, total_output_tokens,
    input_cost, cache_cost, output_cost,
    total_cost_usd,
    ROUND(1 - total_cost_usd / NULLIF(MAX(total_cost_usd) OVER (), 0), 3) AS ahorro_pct
FROM (
    -- Claude REST API
    SELECT 'REST API (Cache)' AS metodo, modelo,
        COUNT(*) AS total_rows,
        SUM(prompt_tokens) AS total_prompt_tokens,
        SUM(NVL(cached_tokens, 0)) AS total_cached_tokens,
        SUM(completion_tokens) AS total_output_tokens,
        ROUND(SUM(input_cost_usd), 4) AS input_cost,
        ROUND(SUM(cache_read_cost_usd), 4) AS cache_cost,
        ROUND(SUM(output_cost_usd), 4) AS output_cost,
        ROUND(SUM(total_cost_usd), 4) AS total_cost_usd
    FROM CORTEX_UTILS.PUBLIC.LLAMADAS_ANALISIS WHERE llm_success GROUP BY modelo
    UNION ALL
    -- Claude AI_COMPLETE
    SELECT 'AI_COMPLETE Native', modelo,
        COUNT(*), SUM(prompt_tokens), NULL, SUM(completion_tokens),
        ROUND(SUM(input_cost_credits) * 4, 4), NULL,
        ROUND(SUM(output_cost_credits) * 4, 4),
        ROUND(SUM(total_cost_usd), 4)
    FROM CORTEX_UTILS.PUBLIC.LLAMADAS_ANALISIS_NATIVE GROUP BY modelo
    UNION ALL
    -- Llama REST API
    SELECT 'REST API (No Cache)', modelo,
        COUNT(*), SUM(prompt_tokens), SUM(NVL(cached_tokens, 0)), SUM(completion_tokens),
        ROUND(SUM(input_cost_usd), 4), ROUND(SUM(cache_read_cost_usd), 4),
        ROUND(SUM(output_cost_usd), 4),
        ROUND(SUM(total_cost_usd), 4)
    FROM CORTEX_UTILS.PUBLIC.LLAMADAS_ANALISIS_LLAMA WHERE llm_success GROUP BY modelo
    UNION ALL
    -- Llama AI_COMPLETE
    SELECT 'AI_COMPLETE Native', modelo,

        ROUND(SUM(input_cost_credits) * 4, 4), NULL,
        ROUND(SUM(output_cost_credits) * 4, 4),
        ROUND(SUM(total_cost_usd), 4)
    FROM CORTEX_UTILS.PUBLIC.LLAMADAS_ANALISIS_NATIVE_LLAMA GROUP BY modelo
) combined
ORDER BY total_cost_usd;

¿Por qué REST API es siempre más barato?

  1. Claude + Cache: 77% de tokens son cache reads a $0.30/M vs $3.00/M.
  2. Llama sin cache: REST $0.72/M vs AI_COMPLETE 0.36cr x $4 = $1.44/M (2x).
  3. Multiplicador créditos: AI_COMPLETE siempre paga $4/cr.
  4. Llama ~10x más barato que Claude en ambos métodos.

Conclusión: Llama REST ($1.00) es el más barato. Claude REST ($2.20) mejor calidad. AI_COMPLETE Claude ($18.56) es 18.5x más caro.
